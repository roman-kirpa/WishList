CREATE PROCEDURE dbo.spSetUserIfNotExist @Name varchar(25)
AS
declare @count int;
select @count = (select count(*) from [dbo].[Users] where Name = @Name);
if @count >0
 INSERT INTO [dbo].[Users] ([Name])
     VALUES (@Name) 

GO
